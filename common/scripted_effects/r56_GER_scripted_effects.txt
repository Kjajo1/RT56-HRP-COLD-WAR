r56_GER_ersatz_idea_upgrade = { 
	if = {
		limit = {
			has_idea = r56_ger_ersatz_idea_3
		}
		swap_ideas = {
			remove_idea = r56_ger_ersatz_idea_3
			add_idea = r56_ger_ersatz_idea_4
		}
	}
	if = {
		limit = {
			has_idea = r56_ger_ersatz_idea_2
		}
		swap_ideas = {
			remove_idea = r56_ger_ersatz_idea_2
			add_idea = r56_ger_ersatz_idea_3
		}
	}
	if = {
		limit = {
			has_idea = r56_ger_ersatz_idea_1
		}
		swap_ideas = {
			remove_idea = r56_ger_ersatz_idea_1
			add_idea = r56_ger_ersatz_idea_2
		}
	}
	if = {
		limit = {
			NOT = {
				OR = {
					has_idea = r56_ger_ersatz_idea_1
					has_idea = r56_ger_ersatz_idea_2
					has_idea = r56_ger_ersatz_idea_3
					has_idea = r56_ger_ersatz_idea_4
				}
			}
		}
		add_ideas = r56_ger_ersatz_idea_1
	}
}
